SERVLET:

import java.io.*;

import java.sql.*;

import javax.servlet.*;

import javax.servlet.http.*;

import javax.servlet.annotation.WebServlet;


@WebServlet("/TicketBookingServlet")

public class TicketBookingServlet extends HttpServlet {

    @Override

    protected void doPost(HttpServletRequest request, HttpServletResponse response)

            throws ServletException, IOException {

        

        response.setContentType("text/html");

        PrintWriter out = response.getWriter();


       

        String name = request.getParameter("passenger_name");

        String movie = request.getParameter("movie_name");

        String date = request.getParameter("show_date");

        String time = request.getParameter("show_time");


        try {

           

            Class.forName("com.mysql.cj.jdbc.Driver");


           

            try (Connection con = DriverManager.getConnection(

                    "jdbc:mysql://localhost:3306/ticket_booking", "root", "")) {

                

               

                String sql = "INSERT INTO movie_tickets (passenger_name, movie_name, show_date, show_time) VALUES (?, ?, ?, ?)";

                PreparedStatement pst = con.prepareStatement(sql);

                pst.setString(1, name);

                pst.setString(2, movie);

                pst.setString(3, date);

                pst.setString(4, time);

                

                int result = pst.executeUpdate();

                

                if (result > 0) {

                    out.println("<h2>Movie ticket booked successfully!</h2>");

                } else {

                    out.println("<h2>Error booking movie ticket.</h2>");

                }

                

               

                Statement stmt = con.createStatement();

                ResultSet rs = stmt.executeQuery("SELECT * FROM movie_tickets");

                

                out.println("<h3> Booked Movie Tickets:</h3>");

                out.println("<table border='1'><tr><th>ID</th><th>Name</th><th>Movie</th><th>Date</th><th>Time</th></tr>");

                while (rs.next()) {

                    out.println("<tr><td>" + rs.getInt("ticket_id") + "</td><td>" +

                            rs.getString("passenger_name") + "</td><td>" +

                            rs.getString("movie_name") + "</td><td>" +

                            rs.getDate("show_date") + "</td><td>" +

                            rs.getString("show_time") + "</td></tr>");

                }

                out.println("</table>");

            }

        } catch (Exception e) {

            e.printStackTrace(out); // show errors in browser

        } finally {

            out.close();

        }

    }

}
